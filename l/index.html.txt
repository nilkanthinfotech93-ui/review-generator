<script>
function generateReview() {

  var service = document.getElementById("service").value;
  var lang = document.getElementById("language").value;

  var reviews_en = [
    "Excellent " + service + " service by Nilkanth Infotech in Ahmedabad. Quick installation and very professional team.",
    "Highly satisfied with " + service + ". Best CCTV service provider in Ahmedabad.",
    "Great experience with Nilkanth Infotech. Reliable and fast service."
  ];

  var reviews_gu = [
    "Nilkanth Infotech દ્વારા ખુબ જ સરસ " + service + " સેવા મળી. ખુબ પ્રોફેશનલ ટીમ.",
    "Ahmedabad માં CCTV માટે શ્રેષ્ઠ સેવા. ખુબ જ સંતોષ થયો.",
    "ખૂબ જ સારી અને ઝડપી સેવા. Highly recommended."
  ];

  var finalReview;

  if(lang == "gu") {
    finalReview = reviews_gu[Math.floor(Math.random() * reviews_gu.length)];
  } else {
    finalReview = reviews_en[Math.floor(Math.random() * reviews_en.length)];
  }

  document.getElementById("output").value = finalReview;
}

function copyText() {
  var text = document.getElementById("output");
  text.select();
  document.execCommand("copy");
  alert("Copied!");
}

function openGoogle() {
  window.open("https://g.page/r/CTjtyeEyrKNIEAE/review");
}
